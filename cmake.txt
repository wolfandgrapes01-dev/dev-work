set(OUTPUT_DIR "${CMAKE_SOURCE_DIR}/cpp_module/build")

foreach(CONFIG IN ITEMS Debug Release RelWithDebInfo MinSizeRel)
    string(TOUPPER ${CONFIG} CONFIG_UP)
    set_target_properties(example PROPERTIES
        LIBRARY_OUTPUT_DIRECTORY_${CONFIG_UP} ${OUTPUT_DIR}
        RUNTIME_OUTPUT_DIRECTORY_${CONFIG_UP} ${OUTPUT_DIR}
        ARCHIVE_OUTPUT_DIRECTORY_${CONFIG_UP} ${OUTPUT_DIR}
    )
endforeach()

set_target_properties(example PROPERTIES
    COMPILE_PDB_NAME "example"
    COMPILE_PDB_OUTPUT_DIRECTORY ${OUTPUT_DIR}
)
